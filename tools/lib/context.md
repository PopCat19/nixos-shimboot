# tools/lib/

- logging.sh — Provides unified logging functions with ANSI colors
- runtime.sh — Provides common runtime helpers for shell scripts
- firmware.sh — Provides firmware management functions for ChromeOS-derived images
- mounts.sh — Provides shared mount and loop device management functions
- devices.sh — Provides device and disk enumeration helpers
- nix.sh — Provides Nix and Cachix interaction helpers
- cleanup.sh — Provides trap-based cleanup helpers
- args.sh — Provides argument parsing utilities
